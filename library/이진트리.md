
### 이진 트리의 순회

```cpp
struct TreeNode {
  TreeNode *left;
  TreeNode *right;
  int data;
};
vector<int> history;
```

위와 같은 형식의 이진트리가 있을때 Update, Search(전위, 중위, 후위) 순회 방법
```cpp
// 전위
void Preorder(TreeNode* root, vector<int> &history) {
  if (!root) return;
  history.push_back(root->data);
  Preorder(root->left, history);
  Preorder(root->right, history);
}
// 중위
void Midorder(TreeNode* root, vector<int> history) {
  if (!root) return;
  Midorder(root->left, history);
  history.push_back(root->data);
  Midorder(root->right, history);
}
// 후위
void Preorder(TreeNode* root, vector<int> history) {
  if (!root) return;
  Preorder(root->left, history);
  Preorder(root->right, history);
  history.push_back(root->data);
}
```

**이진 트리 생성 방법** 
```cpp
void InsertNode(TreeNode *root, TreeNode *newNode) {
    if(newNode->data < root->data) {
        //left 
        if(root->left) {
            InsertNode(root->left, newNode);
        } else {
            root->left = newNode;
        }
    } else {
        //right
        if(root->right) {
            InsertNode(root->right, newNode);
        } else {
            root->right = newNode;
        }
        
    }
}
```
**이진 트리 생성 방법(개선)**
참조 혹은 이중 포인터를 사용해서 로직 간략화를 할수있습니다.
```cpp
void InsertNode(TreeNode*& root, TreeNode *newNode) {
    if( !root ) {
        root = newNode;
        return ;
    }
    if(newNode->data < root->data) {
        InsertNode(root->left, newNode);
    } else {
        InsertNode(root->right, newNode);
    }
}
```
